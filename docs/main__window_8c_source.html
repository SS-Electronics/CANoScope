69const char *gui_find_asset(const char *name)
70{
71 static char buf[1024];
72
73 const char *env = getenv("CANOSCOPE_ASSET_DIR");
74 if (env && *env) {
75 snprintf(buf, sizeof(buf), "%s/%s", env, name);
76 if (access(buf, R_OK) == 0) return buf;
77 }
78
79 char exe[512];
80 ssize_t n = readlink("/proc/self/exe", exe, sizeof(exe) - 1);
81 if (n > 0) {
82 exe[n] = '\0';
83 char *slash = strrchr(exe, '/');
84 if (slash) {
85 *slash = '\0';
86 const char *rel[] = {
87 "assets", "../assets", "../share/canoscope", NULL
88 };
89 for (int i = 0; rel[i]; i++) {
90 snprintf(buf, sizeof(buf), "%s/%s/%s", exe, rel[i], name);
91 if (access(buf, R_OK) == 0) return buf;
92 }
93 }
94 }
95
96 const char *fixed[] = {
97 "assets",
98 "/usr/local/share/canoscope",
99 "/usr/share/canoscope",
100 NULL
101 };
102 for (int i = 0; fixed[i]; i++) {
103 snprintf(buf, sizeof(buf), "%s/%s", fixed[i], name);
104 if (access(buf, R_OK) == 0) return buf;
105 }
106 return NULL;
107}

289static void trace_csv_write_row(FILE *fp, const can_msg_t *msg)
290{
291 char id_buf[16];
292 char data_buf[3 * CANFD_DATA_MAX + 4];
293
294 if (msg->is_extended)
295 snprintf(id_buf, sizeof(id_buf), "%08X", msg->id);
296 else
297 snprintf(id_buf, sizeof(id_buf), "%03X", msg->id);
298
299 if (msg->dlc == 0) {
300 snprintf(data_buf, sizeof(data_buf), "-");
301 } else {
302 size_t pos = 0;
303 for (uint8_t i = 0; i < msg->dlc && i < CANFD_DATA_MAX &&
304 pos + 4 < sizeof(data_buf); i++) {
305 pos += (size_t)snprintf(data_buf + pos, sizeof(data_buf) - pos,
306 i ? " %02X" : "%02X", msg->data[i]);
307 }
308 }
309
310 double ts = (double)msg->timestamp.tv_sec
311 + (double)msg->timestamp.tv_nsec / 1e9;
312
313 fprintf(fp, "%llu,%.6f,%s,%s,%s,%u,%s\n",
314 (unsigned long long)msg->seq,
315 ts,
316 msg->direction == CAN_DIR_TX ? "Tx" : "Rx",
317 id_buf,
318 gui_msg_type_str(msg),
319 msg->dlc,
320 data_buf);
321}

326static void on_trace_save(GtkWidget *w, gpointer d)
327{
328 (void)w; (void)d;
329
330 if (g_app.trace_len == 0) {
331 gui_show_error(g_gui.window, "Save Trace",
332 "No trace data captured yet.\n"
333 "Use Trace > Start, generate some traffic, then Stop.");
334 return;
335 }
336
337 GtkWidget *dlg = gtk_file_chooser_dialog_new(
338 "Save Trace as CSV",
339 GTK_WINDOW(g_gui.window),
340 GTK_FILE_CHOOSER_ACTION_SAVE,
341 "_Cancel", GTK_RESPONSE_CANCEL,
342 "_Save", GTK_RESPONSE_ACCEPT,
343 NULL);
344 gtk_file_chooser_set_do_overwrite_confirmation(
345 GTK_FILE_CHOOSER(dlg), TRUE);
346 gtk_file_chooser_set_current_name(GTK_FILE_CHOOSER(dlg), "trace.csv");
347
348 GtkFileFilter *ff = gtk_file_filter_new();
349 gtk_file_filter_add_pattern(ff, "*.csv");
350 gtk_file_filter_set_name(ff, "CSV trace (*.csv)");
351 gtk_file_chooser_add_filter(GTK_FILE_CHOOSER(dlg), ff);
352
353 if (gtk_dialog_run(GTK_DIALOG(dlg)) == GTK_RESPONSE_ACCEPT) {
354 char *fn = gtk_file_chooser_get_filename(GTK_FILE_CHOOSER(dlg));
355 strncpy(g_app.trace_filename, fn, APP_TRACE_FILE_LEN - 1);
356 g_app.trace_filename[APP_TRACE_FILE_LEN - 1] = '\0';
357 g_free(fn);
358
359 FILE *fp = fopen(g_app.trace_filename, "w");
360 if (fp) {
361 fprintf(fp, "seq,timestamp,direction,id,type,dlc,data\n");
362 pthread_mutex_lock(&g_app.trace_mutex);
363 size_t n = g_app.trace_len;
364 for (size_t i = 0; i < n; i++)
365 trace_csv_write_row(fp, &g_app.trace_buf[i]);
366 pthread_mutex_unlock(&g_app.trace_mutex);
367 fclose(fp);
368 gui_status_message("Saved %zu frame(s) to %s",
369 n, g_app.trace_filename);
370 } else {
371 gui_show_error(g_gui.window, "Trace Error",
372 "Could not open file for writing.");
373 }
374 }
375 gtk_widget_destroy(dlg);
376}

860static int parse_hex_u32_strict(const char *s, uint32_t max, uint32_t *out)
861{
862 if (!s || !out) return -1;
863
864 while (isspace((unsigned char)*s)) s++;
865 if (*s == '\0' || *s == '+' || *s == '-') return -1;
866
867 errno = 0;
868 char *end = NULL;
869 unsigned long v = strtoul(s, &end, 16);
870 if (errno == ERANGE || end == s || v > max) return -1;
871
872 while (isspace((unsigned char)*end)) end++;
873 if (*end != '\0') return -1;
874
875 *out = (uint32_t)v;
876 return 0;
877}

1111static uint8_t canfd_round_len(int n)
1112{
1113 static const uint8_t valid[] = {
1114 0, 1, 2, 3, 4, 5, 6, 7, 8, 12, 16, 20, 24, 32, 48, 64
1115 };
1116 for (size_t i = 0; i < sizeof(valid) / sizeof(valid[0]); i++)

1126static gboolean txp_build_frame(tx_panel_row_t *row, can_msg_t *msg)
1127{
1128 if (!row || !msg)
1129 return FALSE;
1130
1131 memset(msg, 0, sizeof(*msg));
1132
1133 msg->is_extended = gtk_toggle_button_get_active(
1134 GTK_TOGGLE_BUTTON(row->ext_check));
1135 msg->is_remote = gtk_toggle_button_get_active(
1136 GTK_TOGGLE_BUTTON(row->rtr_check));
1137
1138 /* Frames are sent as CAN FD when the link is in FD mode (BRS enabled). */
1139 msg->is_fd = g_app.fd_mode ? 1 : 0;
1140 msg->is_brs = g_app.fd_mode ? 1 : 0;
1141
1142 int req_len = gtk_spin_button_get_value_as_int(
1143 GTK_SPIN_BUTTON(row->dlc_spin));
1144 if (!msg->is_fd && req_len > CAN_MAX_DLC)
1145 req_len = CAN_MAX_DLC;
1146
1147 const char *ids = gtk_entry_get_text(GTK_ENTRY(row->id_entry));
1148 uint32_t id_max = msg->is_extended ? TX_EXT_ID_MAX : TX_STD_ID_MAX;
1149 if (parse_hex_u32_strict(ids, id_max, &msg->id) != 0) {
1150 txp_set_error(row, msg->is_extended
1151 ? "Invalid 29-bit ID"
1152 : "Invalid 11-bit ID");
1153 return FALSE;
1154 }
1155
1156 if (!msg->is_remote) {
1157 for (int i = 0; i < req_len && i < TX_PANEL_DATA_COLS; i++) {
1158 const char *ds = gtk_entry_get_text(
1159 GTK_ENTRY(row->data_entry[i]));
1160 uint32_t value = 0;
1161 if (parse_hex_u32_strict(ds, TX_BYTE_MAX, &value) != 0) {
1162 char err[64];
1163 snprintf(err, sizeof(err), "Invalid data byte [%d]", i);
1164 txp_set_error(row, err);
1165 return FALSE;
1166 }
1167 msg->data[i] = (uint8_t)value;
1168 }
1169 }
1170
1171 /* CAN FD payloads must be a valid length; pad up if needed (extra bytes
1172 * are already zero from the memset above). */
1173 msg->dlc = msg->is_fd ? canfd_round_len(req_len) : (uint8_t)req_len;
1174 return TRUE;
1175}

1181static gboolean txp_queue_frame(int row_index, gboolean cyclic)
1182{
1183 tx_panel_row_t *row = txp_row_from_data(GINT_TO_POINTER(row_index));
1184 if (!row)
1185 return FALSE;
1186
1187 if (!g_app.connected) {
1188 txp_set_error(row, "Not connected");
1189 return FALSE;

1193 if (!msg) {
1194 txp_set_error(row, "Out of memory");
1195 return FALSE;

1199 free(msg);

1201 }

1203 g_async_queue_push(g_app.tx_queue, msg);
1204 if (!cyclic)
1205 txp_set_status(row, "green", "Queued");
1206 return TRUE;
1207}
