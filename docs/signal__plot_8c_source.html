155static int store_find(uint32_t id, int ext, int sig_idx)
156{
157 for (int i = 0; i < s_store.n; i++) {
158 const plot_signal_t *ps = &s_store.sig[i];
159 if (ps->id == id && ps->ext == (ext ? 1 : 0) && ps->sig_idx == sig_idx)
160 return i;
161 }
162 return -1;

165void gui_plot_add_sample(uint32_t id, int is_ext, int sig_idx, double value)
166{
167 if (!s_store.sig)
168 return;
169 int i = store_find(id, is_ext, sig_idx);
170 if (i < 0)
171 return;
172
173 plot_signal_t *ps = &s_store.sig[i];
174 ps->t[ps->head] = plot_now();
175 ps->v[ps->head] = value;
176 ps->head = (ps->head + 1) % PLOT_RING;
177 if (ps->count < PLOT_RING)
178 ps->count++;
179 ps->last = value;
180 ps->has_last = TRUE;
181}

293static double norm_to_y(double norm, double y0, double y1)
294{
295 if (norm < 0.0) norm = 0.0;
296 if (norm > 1.0) norm = 1.0;
297 return y1 - norm * (y1 - y0);
298}

301static void series_scale(const plot_signal_t *sig, double tmin,
302 double *base, double *span)
303{
304 *base = sig->dmin;
305 *span = sig->dmax - sig->dmin;
306 if (*span > 0.0)
307 return;
308 /* Invalid/empty DBC range — auto-scale from the visible samples. */
309 double mn = 1e300, mx = -1e300;
310 for (int k = 0; k < sig->count; k++) {
311 int idx = (sig->head - sig->count + k + PLOT_RING) % PLOT_RING;
312 if (sig->t[idx] < tmin) continue;
313 if (sig->v[idx] < mn) mn = sig->v[idx];
314 if (sig->v[idx] > mx) mx = sig->v[idx];
315 }
316 if (mx > mn) { *base = mn; *span = mx - mn; }
317 else if (mx == mn) { *base = mn - 0.5; *span = 1.0; } /* flat line */
318 else { *base = 0.0; *span = 1.0; } /* no samples yet */
319}

324static gboolean on_draw(GtkWidget *w, cairo_t *cr, gpointer data)
325{
326 plot_window_t *pw = data;
327 GtkAllocation a;
328 gtk_widget_get_allocation(w, &a);
329 const double W = a.width, H = a.height;
330
331 /* Background. */
332 cairo_set_source_rgb(cr, 0.12, 0.12, 0.14);
333 cairo_rectangle(cr, 0, 0, W, H);
334 cairo_fill(cr);
335
336 const double right = 12, top = 18, bottom = 24;
337
338 /* One Y axis per signal, stacked on the left; widen the margin to fit. */
339 const double axis_w = 56.0;
340 int n_axes = pw->n_series;
341 int max_axes = (int)((W * 0.5) / axis_w);
342 if (max_axes < 1) max_axes = 1;
343 if (n_axes > max_axes) n_axes = max_axes;
344 const double left = pw->n_series > 0 ? 6.0 + n_axes * axis_w : 50.0;
345
346 const double x0 = left, x1 = W - right, y0 = top, y1 = H - bottom;
347 if (x1 <= x0 + 10 || y1 <= y0 + 10)
348 return FALSE;
349
350 const double now = win_now(pw);
351 const double winw = pw->window_sec;
352 const double tmin = now - winw;
353
354 cairo_select_font_face(cr, "monospace",
355 CAIRO_FONT_SLANT_NORMAL, CAIRO_FONT_WEIGHT_NORMAL);
356 cairo_set_font_size(cr, 10);
357 cairo_set_line_width(cr, 1.0);
358
359 /* Per-series normalisation (shared by the Y axes and the traces below). */
360 double s_base[PLOT_MAX_SERIES], s_span[PLOT_MAX_SERIES];
361 for (int s = 0; s < pw->n_series; s++)
362 series_scale(&s_store.sig[pw->series[s].sig],
363 tmin, &s_base[s], &s_span[s]);
364
365 /* Horizontal grid (no shared % labels — each signal carries its own axis). */
366 for (int i = 0; i <= 10; i++) {
367 double yy = y0 + (y1 - y0) * i / 10.0;
368 cairo_set_source_rgb(cr, 0.22, 0.22, 0.25);
369 cairo_move_to(cr, x0, yy);
370 cairo_line_to(cr, x1, yy);
371 cairo_stroke(cr);
372 }
373
374 /* Coloured per-signal Y axes with real physical-value tick labels. */
375 cairo_set_font_size(cr, 9);
376 for (int c = 0; c < n_axes; c++) {
377 const win_series_t *ws = &pw->series[c];
378 const plot_signal_t *sig = &s_store.sig[ws->sig];
379 double ax = 2.0 + (double)(c + 1) * axis_w - 8.0; /* axis line x */
380
381 cairo_set_source_rgb(cr, ws->col[0], ws->col[1], ws->col[2]);
382 cairo_set_line_width(cr, 1.2);
383 cairo_move_to(cr, ax, y0);
384 cairo_line_to(cr, ax, y1);
385 cairo_stroke(cr);
386
387 for (int i = 0; i <= 4; i++) {
388 double frac = i / 4.0;
389 double yy = y1 - frac * (y1 - y0);
390 double val = s_base[c] + frac * s_span[c];
391 cairo_move_to(cr, ax - 3, yy);
392 cairo_line_to(cr, ax, yy);
393 cairo_stroke(cr);
394
395 char lbl[24];
396 snprintf(lbl, sizeof(lbl), "%.4g", val);
397 cairo_text_extents_t te;
398 cairo_text_extents(cr, lbl, &te);
399 double lx = ax - 5 - te.width;
400 if (lx < 1) lx = 1;
401 double ly = yy + 3;
402 if (i == 0) ly = yy - 1; /* bottom label: lift inside */
403 else if (i == 4) ly = yy + 8; /* top label: drop inside */
404 cairo_move_to(cr, lx, ly);
405 cairo_show_text(cr, lbl);
406 }
407
408 /* Unit header above the axis (in the signal's colour). */
409 if (sig->unit[0]) {
410 cairo_text_extents_t ue;
411 cairo_text_extents(cr, sig->unit, &ue);
412 double ux = ax - ue.width;
413 if (ux < 1) ux = 1;
414 cairo_move_to(cr, ux, y0 - 6);
415 cairo_show_text(cr, sig->unit);
416 }
417 }
418 cairo_set_font_size(cr, 10);
419
420 /* Vertical grid + X labels (relative seconds). */
421 int xdiv = 10;
422 for (int i = 0; i <= xdiv; i++) {
423 double xx = x0 + (x1 - x0) * i / (double)xdiv;
424 cairo_set_source_rgb(cr, 0.22, 0.22, 0.25);
425 cairo_move_to(cr, xx, y0);
426 cairo_line_to(cr, xx, y1);
427 cairo_stroke(cr);
428 char lbl[16];
429 double rel = -winw + winw * i / (double)xdiv;
430 snprintf(lbl, sizeof(lbl), "%.0fs", rel);
431 cairo_set_source_rgb(cr, 0.6, 0.6, 0.65);
432 cairo_move_to(cr, xx - 8, y1 + 16);
433 cairo_show_text(cr, lbl);
434 }
435
436 /* Plot frame. */
437 cairo_set_source_rgb(cr, 0.4, 0.4, 0.45);
438 cairo_rectangle(cr, x0, y0, x1 - x0, y1 - y0);
439 cairo_stroke(cr);
440
441 if (pw->n_series == 0) {
442 cairo_set_source_rgb(cr, 0.7, 0.7, 0.75);
443 cairo_set_font_size(cr, 13);
444 cairo_move_to(cr, x0 + 14, y0 + 24);
445 cairo_show_text(cr,
446 "Add signals from the dropdown above to plot them over time.");
447 return FALSE;
448 }
449
450 /* Hover hit-test (nearest visible sample across all series). */
451 gboolean have_hit = FALSE;
452 int hit_series = -1;
453 double hit_px = 0, hit_py = 0, hit_t = 0, hit_v = 0, hit_best = 1e300;
454 const gboolean want_hover =
455 pw->hover_on &&
456 pw->hover_x >= x0 && pw->hover_x <= x1 &&
457 pw->hover_y >= y0 && pw->hover_y <= y1;
458
459 /* Traces. */
460 for (int s = 0; s < pw->n_series; s++) {
461 win_series_t *ws = &pw->series[s];
462 const plot_signal_t *sig = &s_store.sig[ws->sig];
463 if (sig->count == 0)
464 continue;
465
466 double base = s_base[s], span = s_span[s];
467
468 /* Find first visible sample + count. */
469 int visible = 0, first = -1;
470 for (int k = 0; k < sig->count; k++) {
471 int idx = (sig->head - sig->count + k + PLOT_RING) % PLOT_RING;
472 if (sig->t[idx] >= tmin) { if (first < 0) first = k; visible++; }
473 }
474 if (visible == 0)
475 continue;
476
477 /* Stride-decimate to ~2 segments per horizontal pixel. */
478 int max_pts = (int)((x1 - x0) * 2.0);
479 int stride = visible > max_pts ? visible / max_pts : 1;
480
481 cairo_set_source_rgb(cr, ws->col[0], ws->col[1], ws->col[2]);
482 cairo_set_line_width(cr, 1.5);
483 gboolean started = FALSE;
484 for (int k = first; k < sig->count; k += stride) {
485 int idx = (sig->head - sig->count + k + PLOT_RING) % PLOT_RING;
486 double tt = sig->t[idx];
487 if (tt < tmin) continue;
488 double x = x0 + (tt - tmin) / winw * (x1 - x0);
489 double y = norm_to_y((sig->v[idx] - base) / span, y0, y1);
490 if (!started) { cairo_move_to(cr, x, y); started = TRUE; }
491 else cairo_line_to(cr, x, y);
492
493 if (want_hover) {
494 double dx = x - pw->hover_x, dy = y - pw->hover_y;
495 double d2 = dx * dx + dy * dy;
496 if (d2 < hit_best) {
497 hit_best = d2;
498 hit_series = s;
499 hit_px = x; hit_py = y;
500 hit_t = tt; hit_v = sig->v[idx];
501 have_hit = TRUE;
502 }
503 }
504 }
505 cairo_stroke(cr);
506 }
507
508 /* Hover crosshair + tooltip. */
509 if (have_hit && hit_best <= PLOT_HOVER_PX * PLOT_HOVER_PX) {
510 win_series_t *ws = &pw->series[hit_series];
511 const plot_signal_t *sig = &s_store.sig[ws->sig];
512
513 /* Crosshair. */
514 double dash[] = { 3.0, 3.0 };
515 cairo_set_dash(cr, dash, 2, 0);
516 cairo_set_line_width(cr, 1.0);
517 cairo_set_source_rgba(cr, 0.8, 0.8, 0.85, 0.6);
518 cairo_move_to(cr, hit_px, y0); cairo_line_to(cr, hit_px, y1);
519 cairo_move_to(cr, x0, hit_py); cairo_line_to(cr, x1, hit_py);
520 cairo_stroke(cr);
521 cairo_set_dash(cr, NULL, 0, 0);
522
523 /* Marker dot in the series colour. */
524 cairo_set_source_rgb(cr, ws->col[0], ws->col[1], ws->col[2]);
525 cairo_arc(cr, hit_px, hit_py, 3.5, 0, 2 * G_PI);
526 cairo_fill(cr);
527
528 /* Tooltip text lines. */
529 char l1[2 * DBC_NAME_MAX + 16];
530 char l2[64];
531 char l3[32];
532 snprintf(l1, sizeof(l1), "%s", sig->label);
533 snprintf(l2, sizeof(l2), "%.5g %s", hit_v, sig->unit);
534 snprintf(l3, sizeof(l3), "t = %.2fs", hit_t - now);
535
536 cairo_set_font_size(cr, 11);
537 cairo_text_extents_t e1, e2, e3;
538 cairo_text_extents(cr, l1, &e1);
539 cairo_text_extents(cr, l2, &e2);
540 cairo_text_extents(cr, l3, &e3);
541 double tw = e1.width;
542 if (e2.width > tw) tw = e2.width;
543 if (e3.width > tw) tw = e3.width;
544 double pad = 6, lh = 15;
545 double bw = tw + 2 * pad, bh = 3 * lh + pad;
546 double bx = hit_px + 12, by = hit_py + 12;
547 if (bx + bw > x1) bx = hit_px - bw - 12;
548 if (by + bh > y1) by = hit_py - bh - 12;
549 if (bx < x0) bx = x0 + 2;
550 if (by < y0) by = y0 + 2;
551
552 cairo_set_source_rgba(cr, 0.05, 0.05, 0.07, 0.92);
553 cairo_rectangle(cr, bx, by, bw, bh);
554 cairo_fill_preserve(cr);
555 cairo_set_source_rgb(cr, ws->col[0], ws->col[1], ws->col[2]);
556 cairo_set_line_width(cr, 1.0);
557 cairo_stroke(cr);
558
559 cairo_set_source_rgb(cr, ws->col[0], ws->col[1], ws->col[2]);
560 cairo_move_to(cr, bx + pad, by + pad + 9);
561 cairo_show_text(cr, l1);
562 cairo_set_source_rgb(cr, 0.92, 0.92, 0.95);
563 cairo_move_to(cr, bx + pad, by + pad + 9 + lh);
564 cairo_show_text(cr, l2);
565 cairo_set_source_rgb(cr, 0.7, 0.7, 0.75);
566 cairo_move_to(cr, bx + pad, by + pad + 9 + 2 * lh);
567 cairo_show_text(cr, l3);
568 }
569
570 return FALSE;
571}

697void gui_plot_set_database(const dbc_db_t *db)
698{
699 /* Rebuild the central store. */
700 g_free(s_store.sig);
701 s_store.sig = NULL;
702 s_store.n = 0;
703
704 if (db && db->signal_count > 0) {
705 s_store.sig = g_malloc0(sizeof(plot_signal_t) * db->signal_count);
706 int n = 0;
707 for (size_t mi = 0; mi < db->message_count; mi++) {
708 const dbc_message_t *m = &db->messages[mi];
709 for (uint16_t si = 0; si < m->signal_count; si++) {
710 const dbc_signal_t *s = &m->signals[si];
711 plot_signal_t *ps = &s_store.sig[n];
712 snprintf(ps->label, sizeof(ps->label), "%s.%s",
713 m->name, s->name);
714 snprintf(ps->unit, sizeof(ps->unit), "%s", s->unit);
715 ps->id = m->id;
716 ps->ext = m->is_extended;
717 ps->sig_idx = si;
718 ps->dmin = s->min;
719 ps->dmax = s->max;
720 n++;
721 }
722 }
723 s_store.n = n;
724 }
725
726 /* The signal set changed: clear every window's selection and refresh its
727 * dropdown so it cannot reference stale store indices. */
728 for (GList *l = s_windows; l; l = l->next) {
729 plot_window_t *pw = l->data;
730 while (pw->n_series > 0)
731 win_remove_series(pw, pw->n_series - 1);
732 pw->color_next = 0;
733 win_populate_combo(pw);
734 if (pw->area)
735 gtk_widget_queue_draw(pw->area);
736 }
737}
