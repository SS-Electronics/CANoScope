86static dbc_message_t *parse_bo(const char *line, dbc_db_t *db)
87{
88 /* BO_ <id> <name>: <dlc> <transmitter> */
89 unsigned long raw_id = 0;
90 char name[DBC_NAME_MAX] = {0};
91 unsigned int dlc = 0;
92
93 if (sscanf(line, "BO_ %lu %63[^:]: %u", &raw_id, name, &dlc) < 2)
94 return NULL;
95
96 /* Trim trailing whitespace from the captured name. */
97 size_t n = strlen(name);
98 while (n > 0 && isspace((unsigned char)name[n - 1]))
99 name[--n] = '\0';
100
101 dbc_message_t *m = dbc_push_message(db);
102 if (!m)
103 return NULL;
104
105 m->is_extended = (raw_id & DBC_EXT_FLAG) ? 1 : 0;
106 m->id = (uint32_t)(raw_id & ~DBC_EXT_FLAG);
107 m->dlc = (uint8_t)(dlc > 64 ? 64 : dlc);
108 snprintf(m->name, sizeof(m->name), "%s", name);
109 return m;
110}

118static int parse_sg(const char *line, dbc_message_t *m)
119{
120 if (!m || m->signal_count >= DBC_MAX_SIGNALS_PER_MSG)
121 return -1;
122
123 /* SG_ <name> : <start>|<len>@<order><sign> (<factor>,<offset>) [<min>|<max>] "<unit>" <recv> */
124 char name[DBC_NAME_MAX] = {0};
125 unsigned start = 0, len = 0;
126 int order = 1; /* 1 = Intel/LE, 0 = Motorola/BE */
127 char sign = '+';
128 double factor = 1.0, offset = 0.0;
129 double minv = 0.0, maxv = 0.0;
130 char unit[DBC_UNIT_MAX] = {0};
131
132 int got = sscanf(line,
133 " SG_ %63s : %u | %u @ %d %c ( %lf , %lf ) [ %lf | %lf ] \"%23[^\"]\"",
134 name, &start, &len, &order, &sign, &factor, &offset,
135 &minv, &maxv, unit);
136
137 /* Require at least up to the byte order; the rest may be absent in terse
138 * databases. (Multiplexed `SG_ name m0 :` forms are skipped — `name`
139 * would capture the multiplex token and the field count falls short.) */
140 if (got < 5)
141 return -1;
142
143 if (len == 0 || len > 64)
144 return -1;
145
146 dbc_signal_t *s = &m->signals[m->signal_count];
147 memset(s, 0, sizeof(*s));
148 snprintf(s->name, sizeof(s->name), "%s", name);
149 s->start_bit = (uint16_t)start;
150 s->length = (uint16_t)len;
151 s->little_endian = order ? 1 : 0;
152 s->is_signed = (sign == '-') ? 1 : 0;
153 s->factor = factor;
154 s->offset = offset;
155 s->min = minv;
156 s->max = maxv;
157 snprintf(s->unit, sizeof(s->unit), "%s", unit);
158
159 m->signal_count++;
160 return 0;
161}

183static void unescape_dbc_string(char *dst, size_t dstsz, const char *src)
184{
185 if (!dst || dstsz == 0)
186 return;
187 dst[0] = '\0';
188 if (!src)
189 return;
190
191 size_t n = 0;
192 for (const char *p = src; *p && n + 1 < dstsz; p++) {
193 if (*p == '\\' && p[1]) {
194 p++;
195 switch (*p) {
196 case 'n': dst[n++] = '\n'; break;
197 case 'r': dst[n++] = '\r'; break;
198 case 't': dst[n++] = '\t'; break;
199 default: dst[n++] = *p; break;
200 }
201 } else {
202 dst[n++] = *p;
203 }
204 }
205 dst[n] = '\0';
206}

235static int parse_dbc_quoted_string(const char *p, char *dst, size_t dstsz)
236{
237 if (!p || !dst || dstsz == 0)
238 return -1;
239 if (*p != '"')
240 return -1;
241
242 p++;
243 size_t n = 0;
244 int closed = 0;
245 for (; *p; p++) {
246 if (*p == '\\' && p[1]) {
247 if (n + 2 < dstsz) {
248 dst[n++] = *p;
249 dst[n++] = p[1];
250 }
251 p++;
252 continue;
253 }
254 if (*p == '"') {
255 closed = 1;
256 break;
257 }
258 if (n + 1 < dstsz)
259 dst[n++] = *p;
260 }
261 dst[n] = '\0';
262 return closed ? 0 : -1;
263}

265static int parse_cm_sg(const char *line, dbc_db_t *db)
266{
267 char name[DBC_NAME_MAX] = {0};
268 char comment[DBC_COMMENT_MAX * 2] = {0};
269
270 const char *p = line;
271 while (isspace((unsigned char)*p))
272 p++;
273 if (strncmp(p, "CM_", 3) != 0)
274 return -1;
275 p += 3;
276 while (isspace((unsigned char)*p))
277 p++;
278 if (strncmp(p, "SG_", 3) != 0)
279 return -1;
280 p += 3;
281 while (isspace((unsigned char)*p))
282 p++;
283
284 char *end = NULL;
285 unsigned long raw_id = strtoul(p, &end, 10);
286 if (end == p)
287 return -1;
288 p = end;
289 while (isspace((unsigned char)*p))
290 p++;
291
292 size_t ni = 0;
293 while (*p && !isspace((unsigned char)*p)) {
294 if (ni + 1 < sizeof(name))
295 name[ni++] = *p;
296 p++;
297 }
298 name[ni] = '\0';
299 while (isspace((unsigned char)*p))
300 p++;
301 if (!name[0] || parse_dbc_quoted_string(p, comment, sizeof(comment)) != 0)
302 return -1;
303
304 dbc_signal_t *sig =
305 find_signal_mut(find_message_by_raw_dbc_id(db, raw_id), name);
306 if (!sig)
307 return -1;
308
309 unescape_dbc_string(sig->comment, sizeof(sig->comment), comment);
310 return 0;
311}

313dbc_db_t *dbc_load_file(const char *path, char *err, size_t errsz)
314{
315 FILE *fp = fopen(path, "r");
316 if (!fp) {
317 if (err) snprintf(err, errsz, "Cannot open '%s'.", path);
318 return NULL;
319 }
320
321 dbc_db_t *db = calloc(1, sizeof(*db));
322 if (!db) {
323 fclose(fp);
324 if (err) snprintf(err, errsz, "Out of memory.");
325 return NULL;
326 }
327 strncpy(db->path, path, sizeof(db->path) - 1);
328
329 char line[1024];
330 dbc_message_t *cur = NULL;
331 while (fgets(line, sizeof(line), fp)) {
332 const char *p = line;
333 while (*p && isspace((unsigned char)*p))
334 p++;
335
336 if (strncmp(p, "BO_ ", 4) == 0) {
337 cur = parse_bo(p, db);
338 } else if (strncmp(p, "SG_ ", 4) == 0) {
339 if (parse_sg(p, cur) == 0)
340 db->signal_count++;
341 } else if (strncmp(p, "CM_ SG_ ", 8) == 0) {
342 parse_cm_sg(p, db);
343 } else if (*p == '\0' || strncmp(p, "BO_TX_BU_", 9) == 0) {
344 /* Blank line or transmitter list — keep the current message. */
345 } else if (!isspace((unsigned char)line[0])) {
346 /* A new top-level (non-indented) record ends the current BO_. */
347 cur = NULL;
348 }
349 }
350 fclose(fp);
351
352 if (db->message_count == 0) {
353 if (err) snprintf(err, errsz,
354 "No CAN messages (BO_) found in '%s'.", path);
355 dbc_free(db);
356 return NULL;
357 }
358 return db;
359}

489 fprintf(fp, "\tBO_TX_BU_\n");
490 fprintf(fp, "\tCM_\n");
491 fprintf(fp, "\tSG_MUL_VAL_\n");
492 fprintf(fp, "\tVAL_\n");
493 fprintf(fp, "\tVAL_TABLE_\n\n");
494 fprintf(fp, "BS_:\n\n");
495 fprintf(fp, "BU_: Vector__XXX\n\n");

499 unsigned long raw_id = (unsigned long)m->id;
500 if (m->is_extended)
501 raw_id |= DBC_EXT_FLAG;

503 fprintf(fp, "BO_ %lu %s: %u Vector__XXX\n",
504 raw_id, m->name, (unsigned)m->dlc);
505
506 for (uint16_t si = 0; si < m->signal_count; si++) {
507 const dbc_signal_t *s = &m->signals[si];
508 char unit[DBC_UNIT_MAX * 2];
509 size_t up = 0;

511 unsigned char c = (unsigned char)s->unit[i];
512 if (c == '"' || c == '\\')
513 continue;
514 unit[up++] = (char)c;
515 }
516 unit[up] = '\0';
517
518 fprintf(fp,
519 " SG_ %s : %u|%u@%u%c (%.15g,%.15g) [%.15g|%.15g] \"%s\" Vector__XXX\n",
520 s->name,
521 (unsigned)s->start_bit,
522 (unsigned)s->length,
523 s->little_endian ? 1u : 0u,
524 s->is_signed ? '-' : '+',
525 s->factor,
526 s->offset,
527 s->min,
528 s->max,
529 unit);
530 }

532 const dbc_signal_t *s = &m->signals[si];
533 if (!s->comment[0])
534 continue;
535 fprintf(fp, "CM_ SG_ %lu %s \"", raw_id, s->name);
536 write_escaped_dbc_string(fp, s->comment);
537 fprintf(fp, "\";\n");
538 }
539 fprintf(fp, "\n");
540 }
541
542 if (fclose(fp) != 0) {

546 }

549 dbc_set_errno(err, errsz, "Cannot replace", path);
550 remove(tmp);
551 return -1;
552 }
553

555}

557uint64_t dbc_extract_raw(const uint8_t *data, uint8_t dlc,
558 const dbc_signal_t *sig)
559{
560 if (!data || !sig || sig->length == 0 || sig->length > 64)
561 return 0;
562
563 uint64_t value = 0;
564 const uint16_t len = sig->length;
565 const uint16_t total = (uint16_t)dlc * 8u;
566
567 if (sig->little_endian) {
568 /* Intel: bit i of the value comes from bit (start_bit + i). */
569 for (uint16_t i = 0; i < len; i++) {
570 uint16_t bit = (uint16_t)(sig->start_bit + i);
571 if (bit >= total)
572 break;
573 uint8_t b = data[bit >> 3];
574 value |= (uint64_t)((b >> (bit & 7)) & 1u) << i;
575 }
576 } else {
577 /* Motorola: consume MSB-first along the sawtooth bit order. */
578 int bit = sig->start_bit;
579 for (uint16_t i = 0; i < len; i++) {
580 if (bit < 0 || bit >= total)
581 break;
582 uint8_t b = data[bit >> 3];
583 value |= (uint64_t)((b >> (bit & 7)) & 1u) << (len - 1 - i);
584 if ((bit & 7) == 0)
585 bit += 15; /* jump to the MSB of the next byte */
586 else
587 bit -= 1;
588 }
589 }
590 return value;
591}

593double dbc_decode_physical(const dbc_signal_t *sig, uint64_t raw,
594 int64_t *signed_raw_out)
595{
596 int64_t sraw = (int64_t)raw;
597
598 if (sig->is_signed && sig->length < 64) {
599 uint64_t sign_bit = (uint64_t)1 << (sig->length - 1);
600 if (raw & sign_bit) {
601 /* Sign-extend by setting all bits above the field width. */
602 uint64_t mask = ~(((uint64_t)1 << sig->length) - 1);
603 sraw = (int64_t)(raw | mask);
604 }
605 }
606
607 if (signed_raw_out)
608 *signed_raw_out = sig->is_signed ? sraw : (int64_t)raw;
609
610 double base = sig->is_signed ? (double)sraw : (double)raw;
611 return base * sig->factor + sig->offset;
612}
