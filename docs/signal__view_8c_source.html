65static void fmt_double(char *buf, size_t sz, double v)
66{
67 snprintf(buf, sz, "%.4f", v);
68 /* Trim trailing zeros and a dangling decimal point. */
69 char *dot = strchr(buf, '.');
70 if (!dot)
71 return;
72 char *end = buf + strlen(buf) - 1;
73 while (end > dot && *end == '0')
74 *end-- = '\0';
75 if (end == dot)
76 *end = '\0';
77}

86static void sig_table_rebuild(void)
87{
88 if (!g_gui.sig_store)
89 return;
90
91 gtk_list_store_clear(g_gui.sig_store);
92 if (!s_db)
93 return;
94
95 for (size_t mi = 0; mi < s_db->message_count; mi++) {
96 const dbc_message_t *m = &s_db->messages[mi];
97 char id_buf[16];
98 gui_format_id(id_buf, sizeof(id_buf), m->id, m->is_extended);
99
100 for (uint16_t si = 0; si < m->signal_count; si++) {
101 const dbc_signal_t *s = &m->signals[si];
102 char range[80], minb[32], maxb[32];
103 fmt_double(minb, sizeof(minb), s->min);
104 fmt_double(maxb, sizeof(maxb), s->max);
105 snprintf(range, sizeof(range), "%s .. %s", minb, maxb);
106
107 GtkTreeIter it;
108 gtk_list_store_append(g_gui.sig_store, &it);
109 gtk_list_store_set(g_gui.sig_store, &it,
110 SCOL_MSG, m->name,
111 SCOL_ID, id_buf,
112 SCOL_SIGNAL, s->name,
113 SCOL_RAW, "—",
114 SCOL_VALUE, "—",
115 SCOL_UNIT, s->unit,
116 SCOL_RANGE, range,
117 SCOL_COUNT, 0u,
118 SCOL_FG, NULL,
119 SCOL_MSGID, m->id,
120 SCOL_EXT, (gboolean)m->is_extended,
121 SCOL_SIGIDX, (guint)si,
122 -1);
123 }
124 }
125}

152gboolean gui_signal_load_dbc(const char *path)
153{
154 if (!path || !*path)
155 return FALSE;
156
157 char err[256] = {0};
158 dbc_db_t *db = dbc_load_file(path, err, sizeof(err));
159 if (!db) {
160 gui_status_message("DBC load failed: %s", err);
161 return FALSE;
162 }
163
164 if (s_db)
165 dbc_free(s_db);
166 s_db = db;
167
168 sig_table_rebuild();
169 gui_plot_set_database(s_db);
170 gui_math_set_database(s_db);
171 sig_update_dbc_label();
172 gui_status_message("Loaded database %s (%zu messages, %zu signals).",
173 db->path, db->message_count, db->signal_count);
174 return TRUE;
175}

208void gui_signal_decode_message(const can_msg_t *msg)
209{
210 if (!s_db || !g_gui.sig_store || !msg || msg->is_error)
211 return;
212
213 const dbc_message_t *m = dbc_find_message(s_db, msg->id, msg->is_extended);
214 if (!m)
215 return;
216
217 GtkTreeIter it;
218 if (!gtk_tree_model_get_iter_first(GTK_TREE_MODEL(g_gui.sig_store), &it))
219 return;
220
221 do {
222 guint row_id = 0, sig_idx = 0, cnt = 0;
223 gboolean row_ext = FALSE;
224 gtk_tree_model_get(GTK_TREE_MODEL(g_gui.sig_store), &it,
225 SCOL_MSGID, &row_id,
226 SCOL_EXT, &row_ext,
227 SCOL_SIGIDX, &sig_idx,
228 SCOL_COUNT, &cnt,
229 -1);
230
231 if (row_id != m->id ||
232 row_ext != (gboolean)m->is_extended ||
233 sig_idx >= m->signal_count)
234 continue;
235
236 const dbc_signal_t *s = &m->signals[sig_idx];
237 int64_t sraw = 0;
238 uint64_t raw = dbc_extract_raw(msg->data, msg->dlc, s);
239 double phys = dbc_decode_physical(s, raw, &sraw);
240
241 /* Feed the time-graph viewer with this decoded sample. */
242 gui_plot_add_sample(m->id, m->is_extended, (int)sig_idx, phys);
243 gui_math_add_sample(m->id, m->is_extended, (int)sig_idx, phys);
244
245 char raw_buf[32], val_buf[40];
246 snprintf(raw_buf, sizeof(raw_buf), "%lld", (long long)sraw);
247 fmt_double(val_buf, sizeof(val_buf), phys);
248
249 gtk_list_store_set(g_gui.sig_store, &it,
250 SCOL_RAW, raw_buf,
251 SCOL_VALUE, val_buf,
252 SCOL_COUNT, cnt + 1,
253 SCOL_FG, msg->direction == CAN_DIR_TX ? "blue" : "black",
254 -1);
255 } while (gtk_tree_model_iter_next(GTK_TREE_MODEL(g_gui.sig_store), &it));
256}
