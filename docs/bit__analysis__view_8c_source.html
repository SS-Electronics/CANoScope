191static uint64_t payload_hash(const uint8_t *data, uint8_t dlc)
192{
193 uint64_t h = 1469598103934665603ull;
194 h ^= dlc;
195 h *= 1099511628211ull;
196 for (uint8_t i = 0; i < dlc; i++) {
197 h ^= data[i];
198 h *= 1099511628211ull;
199 }
200 return h;

233static void update_survey(const can_msg_t *msg)
234{
235 if (!msg || msg->is_error || msg->is_remote)
236 return;
237
238 ba_survey_row_t *r = find_survey_row(msg->id,
239 msg->is_extended ? 1u : 0u,
240 msg->direction);
241 if (!r)
242 return;
243
244 int64_t ts = ba_timespec_to_ns(&msg->timestamp);
245 if (r->count == 0)
246 r->first_ts_ns = ts;
247 if (r->last_ts_ns > 0 && ts > r->last_ts_ns) {
248 double dt_ms = (double)(ts - r->last_ts_ns) / 1e6;
249 uint64_t k = r->count;
250 double delta = dt_ms - r->mean_period_ms;
251 r->mean_period_ms += delta / (double)k;
252 r->m2_period_ms += delta * (dt_ms - r->mean_period_ms);
253 if (r->min_period_ms == 0.0 || dt_ms < r->min_period_ms)
254 r->min_period_ms = dt_ms;
255 if (dt_ms > r->max_period_ms)
256 r->max_period_ms = dt_ms;
257 }
258 r->last_ts_ns = ts;
259
260 uint8_t dlc = msg->dlc > CANFD_DATA_MAX ? CANFD_DATA_MAX : msg->dlc;
261 if (r->have_prev) {
262 if (dlc != r->dlc || memcmp(r->prev_data, msg->data, dlc) != 0)
263 r->change_count++;
264 uint8_t max = dlc > r->dlc ? dlc : r->dlc;
265 if (max > BA_CLASSIC_MAX_BYTES)
266 max = BA_CLASSIC_MAX_BYTES;
267 for (uint8_t b = 0; b < max; b++) {
268 uint8_t diff = r->prev_data[b] ^ msg->data[b];
269 for (uint8_t bit = 0; bit < 8; bit++)
270 if (diff & (uint8_t)(1u << bit))
271 r->changed_mask |= (uint64_t)1 << (b * 8u + bit);
272 }
273 }
274
275 uint64_t h = payload_hash(msg->data, dlc);
276 int known = 0;
277 for (uint32_t i = 0; i < r->unique_count; i++) {
278 if (r->unique_hashes[i] == h) {
279 known = 1;
280 break;
281 }
282 }
283 if (!known) {
284 if (r->unique_count < BA_UNIQUE_TRACK)
285 r->unique_hashes[r->unique_count++] = h;
286 else
287 r->unique_overflow = 1;
288 }
289
290 r->count++;
291 r->is_fd = msg->is_fd ? 1u : 0u;
292 r->dlc = dlc;
293 memcpy(r->data, msg->data, dlc);
294 memcpy(r->prev_data, msg->data, dlc);
295 r->have_prev = 1;
296 s_ba.dirty_survey = TRUE;
297}

774{
775 (void)button;
776 (void)data;
777 if (s_ba.analysis_running) {
778 set_status("Analysis is already running.");
779 return;
780 }
781 if (s_ba.session.sample_count < 2) {
782 set_status("Need at least two target frames before analysis.");

785
786 analysis_work_t *work = g_new0(analysis_work_t, 1);
787 work->generation = ++s_ba.analysis_generation;
788 work->work_session = s_ba.session;
789 size_t n = s_ba.session.sample_count;
790 if (n > BA_UI_ANALYZE_MAX_SAMPLES)
791 n = BA_UI_ANALYZE_MAX_SAMPLES;
792 if (ba_session_init(&work->work_session, n ? n : 1) != 0) {
793 g_free(work);

819 snprintf(ws->input_name, sizeof(ws->input_name), "%s", input_name);
820 snprintf(ws->input_unit, sizeof(ws->input_unit), "%s", input_unit);

822 ws->baseline_sample_count = s_ba.session.baseline_sample_count;

841 for (size_t i = 0; i < n; i++) {
842 int src = (int)((s_ba.session.sample_head +

1161static double cell_score_for_bit(uint16_t bit)
1162{
1163 if (bit >= BA_FD_MAX_BITS)
1164 return 0.0;
1165 const ba_bit_stats_t *st = &s_ba.session.bit_stats[bit];
1166 ba_view_mode_t mode = gtk_combo_box_get_active(GTK_COMBO_BOX(s_ba.mode_combo));
1167 switch (mode) {
1168 case BA_VIEW_BASELINE:
1169 return s_ba.session.baseline_valid ?
1170 (double)s_ba.session.baseline_bits[bit] : 0.0;
1171 case BA_VIEW_XOR:
1172 return s_ba.session.baseline_valid ?
1173 (double)(ba_get_bit(s_ba.session.latest_data, bit) ^
1174 s_ba.session.baseline_bits[bit]) : 0.0;
1175 case BA_VIEW_ACTIVITY:
1176 return fmin((double)st->flip_count / 20.0, 1.0);
1177 case BA_VIEW_ENTROPY:
1178 return st->entropy_bits;
1179 case BA_VIEW_CORRELATION:
1180 return fmax(fmax(fabs(st->pearson_correlation),
1181 fabs(st->lagged_score)),
1182 st->state_separation);
1183 case BA_VIEW_LIVE:
1184 default:
1185 return ba_get_bit(s_ba.session.latest_data, bit) ? 1.0 : 0.0;
1186 }
1187}
