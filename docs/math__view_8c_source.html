128static double corr_for_lag(const double *x, const double *y, int n, int lag)
129{
130 int start_x = lag > 0 ? 0 : -lag;
131 int start_y = lag > 0 ? lag : 0;
132 int m = n - abs(lag);
133 if (m < 3)
134 return 0.0;
135
136 double sx = 0.0, sy = 0.0;
137 for (int i = 0; i < m; i++) {
138 sx += x[start_x + i];
139 sy += y[start_y + i];
140 }
141 double mx = sx / m;
142 double my = sy / m;
143
144 double cov = 0.0, vx = 0.0, vy = 0.0;
145 for (int i = 0; i < m; i++) {
146 double dx = x[start_x + i] - mx;
147 double dy = y[start_y + i] - my;
148 cov += dx * dy;
149 vx += dx * dx;
150 vy += dy * dy;
151 }
152 if (vx <= 0.0 || vy <= 0.0)
153 return 0.0;
154 return cov / sqrt(vx * vy);
155}

164static void compute_ranks(const double *v, int n, double *rank)
165{
166 rank_item_t *items = g_new(rank_item_t, n);
167 for (int i = 0; i < n; i++) {
168 items[i].v = v[i];
169 items[i].idx = i;
170 }
171 qsort(items, n, sizeof(*items), cmp_rank_item);
172
173 int i = 0;
174 while (i < n) {
175 int j = i + 1;
176 while (j < n && items[j].v == items[i].v)
177 j++;
178 double r = ((double)i + (double)j + 1.0) / 2.0;
179 for (int k = i; k < j; k++)
180 rank[items[k].idx] = r;
181 i = j;
182 }
183 g_free(items);
184}

231static math_stats_t compute_stats(const double *x, const double *y, int n)
232{
233 math_stats_t st;
234 memset(&st, 0, sizeof(st));
235 st.n = n;
236 if (n <= 0)
237 return st;
238
239 st.min_x = st.max_x = x[0];
240 st.min_y = st.max_y = y[0];
241 for (int i = 0; i < n; i++) {
242 st.mean_x += x[i];
243 st.mean_y += y[i];
244 if (x[i] < st.min_x) st.min_x = x[i];
245 if (x[i] > st.max_x) st.max_x = x[i];
246 if (y[i] < st.min_y) st.min_y = y[i];
247 if (y[i] > st.max_y) st.max_y = y[i];
248 }
249 st.mean_x /= n;
250 st.mean_y /= n;
251
252 double sx = 0.0, sy = 0.0, cov_sum = 0.0;
253 for (int i = 0; i < n; i++) {
254 double dx = x[i] - st.mean_x;
255 double dy = y[i] - st.mean_y;
256 sx += dx * dx;
257 sy += dy * dy;
258 cov_sum += dx * dy;
259 }
260
261 if (n > 1) {
262 st.std_x = sqrt(sx / (n - 1));
263 st.std_y = sqrt(sy / (n - 1));
264 st.cov = cov_sum / (n - 1);
265 }
266 if (sx > 0.0 && sy > 0.0)
267 st.pearson = cov_sum / sqrt(sx * sy);
268
269 if (n > 2) {
270 double *rx = g_new(double, n);
271 double *ry = g_new(double, n);
272 compute_ranks(x, n, rx);
273 compute_ranks(y, n, ry);
274 st.spearman = corr_for_lag(rx, ry, n, 0);
275 g_free(rx);
276 g_free(ry);
277 }
278
279 if (sx > 0.0) {
280 st.slope = cov_sum / sx;
281 st.intercept = st.mean_y - st.slope * st.mean_x;
282
283 double sse = 0.0, mae = 0.0;
284 for (int i = 0; i < n; i++) {
285 double pred = st.intercept + st.slope * x[i];
286 double e = y[i] - pred;
287 sse += e * e;
288 mae += fabs(e);
289 }
290 st.mae = mae / n;
291 st.rmse = sqrt(sse / n);
292 st.r2 = sy > 0.0 ? 1.0 - (sse / sy) : 0.0;
293 }
294
295 st.best_lag = 0;
296 st.best_lag_corr = st.pearson;
297 int max_lag = n / 4;
298 if (max_lag > 20)
299 max_lag = 20;
300 for (int lag = -max_lag; lag <= max_lag; lag++) {
301 double c = corr_for_lag(x, y, n, lag);
302 if (fabs(c) > fabs(st.best_lag_corr)) {
303 st.best_lag_corr = c;
304 st.best_lag = lag;
305 }
306 }
307 return st;
308}
