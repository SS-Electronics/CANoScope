42static void populate_interfaces(GtkComboBoxText *combo)
43{
44#define MAX_IF 32
45 char names[MAX_IF][SOCKETCAN_MAX_IFACE];
46 int n = socketcan_list_interfaces(names, MAX_IF);
47
48 /* Start from a clean list so this is safe to call again from Refresh. */
49 gtk_combo_box_text_remove_all(combo);
50
51 /* Always add vcan0 as the first / fallback option */
52 gtk_combo_box_text_append_text(combo, "vcan0");
53
54 for (int i = 0; i < n; i++) {
55 if (strcmp(names[i], "vcan0") != 0)
56 gtk_combo_box_text_append_text(combo, names[i]);
57 }
58
59 /* Pre-select the interface currently stored in g_app */
60 GtkTreeModel *model =
61 gtk_combo_box_get_model(GTK_COMBO_BOX(combo));
62 GtkTreeIter iter;
63 gboolean valid =
64 gtk_tree_model_get_iter_first(model, &iter);
65 int idx = 0, sel = 0;
66 while (valid) {
67 gchar *txt = NULL;
68 gtk_tree_model_get(model, &iter, 0, &txt, -1);
69 if (txt && strcmp(txt, g_app.iface) == 0) sel = idx;
70 g_free(txt);
71 valid = gtk_tree_model_iter_next(model, &iter);
72 idx++;
73 }
74 gtk_combo_box_set_active(GTK_COMBO_BOX(combo), sel);
75#undef MAX_IF
76}

152static uint32_t parse_bitrate(const char *s)
153{
154 if (!s) return 0;
155 while (*s == ' ') s++;
156 char *end = NULL;
157 double v = strtod(s, &end);
158 if (end == s || v <= 0.0)
159 return 0;
160 while (*end == ' ') end++;
161 if (*end == 'M' || *end == 'm')
162 v *= 1e6;
163 else if (*end == 'k' || *end == 'K')
164 v *= 1e3;
165 else if (v < 10000.0)
166 v *= 1e3; /* bare small number → interpret as kbit/s */
167 return (uint32_t)(v + 0.5);
168}
